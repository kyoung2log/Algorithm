```python
from collections import deque

n, m = map(int, input().split())
j = int(input())

q = deque()
for i in range(n):
    q.append(1) if i<m else q.append(0)

def check(d):
    if q[c-1] == 1:
        return 0
    else:
        q.rotate(d)
        return 1 +  check(d)

prev = 1
result = 0
for _ in range(j):
    c = int(input())
    if prev < c:
        result += check(1)
    else:
        result += check(-1)
    prev = c

print(result)
```
